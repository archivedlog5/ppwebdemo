# JSSDK v6 Demo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 20 PayPal JSSDK v6 demos under `/paypal/jssdk-v6/`, mirroring the jssdk-v5 structure but using the v6 async `createInstance` API, web-component buttons, and `session.start()` pattern.

**Architecture:** v6-specific `_factory.js` handles route generation and PayPal REST API calls; shared `init.js` exposes `getPPInstance()` singleton; each product has its own EJS + public JS (ECM/ECS pairs share one JS file where the SDK call is identical). The PayPal SDK (`web-sdk/v6/core`) is loaded via `<script defer>` in each EJS body — not from the shared header — so `window.DEMO` config is injected by EJS and the SDK interaction lives in `public/js/paypal/jssdk-v6/`.

**Tech Stack:** Node.js + Express, EJS, Vanilla JS (ES5-style IIFE), PayPal REST v2 Orders API (no server SDK), Supabase (demohub.products table)

**Design spec:** `apps/demo-hub/docs/superpowers/specs/2026-05-30-jssdk-v6-design.md`

**Files:**

**v6 vs v5 differences in factory:**
- `SDK_VERSION = 'jssdk-v6'` (not `'jssdk-v5'`)
- GET handler injects `clientId` (not `sdkUrl`) — used by `init.js`'s `createInstance`
- `create-order` returns `res.json({ orderId: order.id })` (lowercase d, object)
- `capture-order` reads `const { orderId } = req.body` (lowercase d)
- No `sdkUrl` construction, no `sdkParams` in GET handler

- [ ] **Step 1: Create `_factory.js`**

```js
// apps/demo-hub/src/routes/paypal/jssdk-v6/_factory.js
'use strict'

const express = require('express')
const fetch = require('node-fetch')
const {
  INTENT,
  validateAmount,
  isZeroDecimal,
  DEFAULT_AMOUNT,
  DEFAULT_CURRENCY,
  SUPPORTED_CURRENCIES,
} = require('../../../config/constants')
const demoParams = require('../../../config/demoParams')
const { getProviderProducts } = require('../../../config/productRegistry')

const SDK_VERSION = 'jssdk-v6'
const PROVIDER = 'paypal'
const API = 'https://api-m.sandbox.paypal.com'

function getHeaders(token) {
  return {
    'Content-Type': 'application/json',
    Authorization: `Bearer ${token}`,
  }
}

async function getAccessToken() {
  const creds = Buffer.from(
    `${process.env.PAYPAL_CN_CLIENT_ID}:${process.env.PAYPAL_CN_SECRET}`
  ).toString('base64')
  const r = await fetch(`${API}/v1/oauth2/token`, {
    method: 'POST',
    headers: {
      Authorization: `Basic ${creds}`,
      'Content-Type': 'application/x-www-form-urlencoded',
    },
    body: 'grant_type=client_credentials',
  })
  const d = await r.json()
  if (!d.access_token) throw new Error('Failed to get access token: ' + JSON.stringify(d))
  return d.access_token
}

async function getCNToken() {
  return getAccessToken()
}

/**
 * createStandardRoute — factory for simple ECM/ECS products.
 *
 * @param {object} opts
 * @param {string} opts.productKey   — e.g. 'paypal-ecm'
 * @param {string} opts.view         — EJS view path, e.g. 'paypal/jssdk-v6/paypal-ecm'
 * @param {function} opts.buildBody  — (amount, currency) => PayPal order request body object
 */
function createStandardRoute({ productKey, view, buildBody }) {
  const router = express.Router()
  const currentProductKey = productKey
  const currentSdkVersion = SDK_VERSION

  // GET — render demo page
  router.get(`/${productKey}`, async (req, res, next) => {
    try {
      const currency = (req.query.currency || DEFAULT_CURRENCY).toUpperCase()
      const amount = req.query.amount || DEFAULT_AMOUNT

      res.render(view, {
        title: productKey,
        provider: PROVIDER,
        sdkVersion: SDK_VERSION,
        currentProductKey,
        currentSdkVersion,
        sidebarProducts: getProviderProducts(PROVIDER),
        showSidebar: true,
        clientId: process.env.PAYPAL_CN_CLIENT_ID,
        defaultAmount: amount,
        currency,
        supportedCurrencies: SUPPORTED_CURRENCIES,
        country: req.query.country || '',
      })
    } catch (err) {
      next(err)
    }
  })

  // POST create-order
  router.post(`/api/${productKey}/create-order`, async (req, res) => {
    try {
      const { amount, currency } = req.body
      const err = validateAmount(amount, currency)
      if (err) return res.status(400).json({ error: err })

      const token = await getCNToken()
      const body = buildBody(amount, currency)
      const r = await fetch(`${API}/v2/checkout/orders`, {
        method: 'POST',
        headers: getHeaders(token),
        body: JSON.stringify(body),
      })
      const order = await r.json()
      if (!order.id) return res.status(500).json({ error: order.message || 'create-order failed' })
      res.json({ orderId: order.id })
    } catch (err) {
      res.status(500).json({ error: err.message })
    }
  })

  // POST capture-order
  router.post(`/api/${productKey}/capture-order`, async (req, res) => {
    try {
      const { orderId } = req.body
      if (!orderId) return res.status(400).json({ error: 'orderId required' })

      const token = await getCNToken()
      const r = await fetch(`${API}/v2/checkout/orders/${orderId}/capture`, {
        method: 'POST',
        headers: getHeaders(token),
      })
      const order = await r.json()
      res.json(order)
    } catch (err) {
      res.status(500).json({ error: err.message })
    }
  })

  return router
}

module.exports = { createStandardRoute, getCNToken, getHeaders, getAccessToken }
```

;(function () {
  'use strict'
  var _promise = null

  window.getPPInstance = function () {
    if (_promise) return _promise
    _promise = window.paypal
      .createInstance({
        clientId: window.DEMO.clientId,
        components: window.DEMO.components || ['paypal-payments'],
      })
      .then(function (inst) {
        try {
          sessionStorage.setItem('pp_v6_clientId', window.DEMO.clientId)
        } catch (e) {}
        return inst
      })
    return _promise
  }
})()

## Task 8: Shared `paypal.js` (ECM + ECS)

**Files:**
- Create: `apps/demo-hub/src/public/js/paypal/jssdk-v6/paypal.js`

This file is shared by `paypal-ecm.ejs` and `paypal-ecs.ejs`. The SDK calls are identical; the order body difference (ECM vs ECS shipping) is handled entirely in the backend route's `buildBody`.

Critical patterns:
- `window.addEventListener('load', ...)` — safe because SDK loaded with `defer`
- `getPPInstance()` — from `init.js`, cached Promise
- `findEligibleMethods().isEligible('paypal')` — v6 eligibility check
- `document.createElement('paypal-button')` — v6 web component
- `instance.createPayPalOneTimePaymentSession({ onApprove, onCancel, onError })`
- `session.start({ presentationMode: 'auto' }, orderPromise)` — **orderPromise is a Promise, never awaited**
- Capture success check: `captures[0].status === 'COMPLETED'`

- [ ] **Step 1: Create `paypal.js`**

```js
// apps/demo-hub/src/public/js/paypal/jssdk-v6/paypal.js
;(function () {
  'use strict'

  function getCurrency() {
    var sel = document.getElementById('demo-currency')
    return sel ? sel.value : 'USD'
  }

  function getAmount() {
    var inp = document.getElementById('demo-amount')
    return inp ? inp.value.trim() : ''
  }

  function validateAmount() {
    var amt = parseFloat(getAmount())
    if (!amt || amt <= 0) {
      showResult('Please enter a valid amount.', 'error')
      return false
    }
    return true
  }

  function clearLoading() {
    var container = document.getElementById('paypal-button-container')
    if (container) container.innerHTML = ''
    return container
  }

  function showResult(text, type) {
    var el = document.getElementById('result-message')
    if (!el) return
    el.textContent = text
    el.className = 'result-message result-' + (type || 'info')
    el.style.display = 'block'
  }

  // Currency change → reload page with new currency param
  document.addEventListener('DOMContentLoaded', function () {
    var sel = document.getElementById('demo-currency')
    if (!sel) return
    sel.addEventListener('change', function () {
      var url = new URL(window.location.href)
      url.searchParams.set('currency', this.value)
      var amt = document.getElementById('demo-amount')
      if (amt) url.searchParams.set('amount', amt.value.trim())
      window.location.replace(url.toString())
    })
  })

  window.addEventListener('load', function () {
    if (typeof paypal === 'undefined') {
      showResult('✗ PayPal SDK failed to load', 'error')
      return
    }

    var urls = window.DEMO.urls

    getPPInstance()
      .then(function (instance) {
        return instance.findEligibleMethods()
      })
      .then(function (eligibility) {
        if (!eligibility.isEligible('paypal')) {
          showResult('PayPal not eligible in this region', 'error')
          return
        }

        var container = clearLoading()
        var btn = document.createElement('paypal-button')
        container.appendChild(btn)

        return instance
          .createPayPalOneTimePaymentSession({
            onApprove: function (data) {
              return fetch(urls.captureOrder, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ orderId: data.orderId }),
              })
                .then(function (r) { return r.json() })
                .then(function (order) {
                  var capture =
                    order.purchase_units &&
                    order.purchase_units[0] &&
                    order.purchase_units[0].payments &&
                    order.purchase_units[0].payments.captures &&
                    order.purchase_units[0].payments.captures[0]
                  if (!capture || capture.status !== 'COMPLETED') {
                    showResult(
                      '✗ Capture failed · status: ' + (capture ? capture.status : 'unknown'),
                      'error'
                    )
                    return
                  }
                  showResult('✓ Payment captured · Order: ' + order.id, 'success')
                })
            },
            onCancel: function () {
              showResult('Payment cancelled.', 'error')
            },
            onError: function (err) {
              showResult('✗ ' + (err.message || String(err)), 'error')
            },
          })
          .then(function (session) {
            btn.addEventListener('click', function () {
              if (!validateAmount()) return
              var orderPromise = fetch(urls.createOrder, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ amount: getAmount(), currency: getCurrency() }),
              })
                .then(function (r) { return r.json() })
                .then(function (d) {
                  if (d.error) throw new Error(d.error)
                  return { orderId: d.orderId }
                })
              session.start({ presentationMode: 'auto' }, orderPromise)
            })
          })
      })
      .catch(function (err) {
        showResult('✗ ' + err.message, 'error')
      })
  })
})()
```
