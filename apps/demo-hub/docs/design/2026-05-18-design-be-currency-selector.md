## 概述

在动态金额输入框旁边加入币种下拉选择器，支持 PayPal 平台的 30 种货币。

**EJS 页面**（demo-hub）：切换币种触发页面刷新，SDK 以新货币重新加载。
**React 页面**（store-fashion，未来）：动态切换，不刷新，后续单独设计。

## 技术方案：URL 参数 + 页面刷新

### 切换流程

```
用户选择 EUR
  │
  ▼ JS onchange 事件
  URL 构建：当前路径 + ?currency=EUR&amount=<当前金额>
  location.replace(newUrl)
  │
  ▼ 页面刷新
  GET /paypal/jssdk-v5/spb-ecm?currency=EUR&amount=100.00
  │
  ▼ Express GET handler
  req.query.currency = 'EUR'（校验在 SUPPORTED_CURRENCIES 内）
  req.query.amount = '100.00'（填充输入框默认值）
  │
  ▼ EJS 渲染
  sdkUrl = '...&currency=EUR'     ← SDK 以 EUR 加载
  window.DEMO.currency = 'EUR'    ← 前端 JS 读取
  defaultAmount = '100.00'        ← 输入框默认值
```

## Part 1：`config/constants.js` 更新

### 新增常量

```js
// 支持的货币列表（30 种）
const SUPPORTED_CURRENCIES = [
  'USD','EUR','GBP','CAD','AUD','BRL','CHF','CZK','DKK',
  'HKD','HUF','ILS','JPY','MXN','NOK','NZD','PHP','PLN',
  'SEK','SGD','THB','TWD','KRW','CLP','SAR','UYU','COP',
  'IDR','PEN','AED'
]

// 零小数位货币（整数金额，无 cents）
const ZERO_DECIMAL_CURRENCIES = new Set(['JPY','KRW','TWD','CLP','IDR'])

function isZeroDecimal(currency) {
  return ZERO_DECIMAL_CURRENCIES.has(currency)
}
```

### `buildOrderBody` 更新

`currency` 参数现在通过 `overrides.currency` 传入，金额按币种格式化：

```js
function buildOrderBody(amount, overrides = {}) {
  const currency = SUPPORTED_CURRENCIES.includes(overrides.currency)
    ? overrides.currency
    : DEFAULT_CURRENCY

  // 零小数位货币取整，其余两位小数
  const num   = parseFloat(amount)
  const value = isZeroDecimal(currency)
    ? String(Math.round(num))
    : num.toFixed(2)

  // ... purchase_units, items, shipping 不变，currency_code 使用 currency 变量
}
```

### 服务端金额校验更新

零小数位货币的最小值校验保持 1（整数），最大值 30000 不变：

```js
function validateAmount(amount, currency = DEFAULT_CURRENCY) {
  const num = parseFloat(amount)
  if (!amount || isNaN(num)) return 'Invalid amount'
  if (num < MIN_AMOUNT) return `Amount must be at least ${MIN_AMOUNT}`
  if (num > MAX_AMOUNT) return `Amount cannot exceed ${MAX_AMOUNT.toLocaleString()}`
  if (!isZeroDecimal(currency) && !/^\d+(\.\d{1,2})?$/.test(String(amount))) {
    return 'Amount must have at most 2 decimal places'
  }
  if (isZeroDecimal(currency) && !/^\d+$/.test(String(Math.round(num)))) {
    return `${currency} does not support decimal amounts`
  }
  return null
}
```

## Part 4：前端 JS 更新

### `getCurrency()` + 页面刷新逻辑

每个 JS 文件（spb.js、acdc.js、buttons.js、vault-return.js）新增：

```js
function getCurrency() {
  var sel = document.getElementById('demo-currency')
  return sel ? sel.value : (window.DEMO && window.DEMO.currency) || 'USD'
}

function isZeroDecimal(currency) {
  var zd = ['JPY','KRW','TWD','CLP','IDR']
  return zd.indexOf(currency) !== -1
}

// 切换币种时刷新页面（携带当前金额保留）
document.addEventListener('DOMContentLoaded', function () {
  var currencySel = document.getElementById('demo-currency')
  if (!currencySel) return
  currencySel.addEventListener('change', function () {
    var amount   = document.getElementById('demo-amount')
    var amtVal   = amount ? amount.value.trim() : ''
    var url      = new URL(window.location.href)
    url.searchParams.set('currency', this.value)
    if (amtVal) url.searchParams.set('amount', amtVal)
    window.location.replace(url.toString())
  })
})
```

### `validateAmount` 零小数位适配

```js
var currency = getCurrency()
// blur 时：零小数位取整，其他两位小数
amountInput.addEventListener('blur', function () {
  var num = parseFloat(this.value)
  if (!isNaN(num) && num > 0) {
    this.value = isZeroDecimal(currency) ? String(Math.round(num)) : num.toFixed(2)
  }
  validateAmount()
})

// validateAmount 里额外检查：零小数位不允许小数
function validateAmount() {
  var val = input.value.trim()
  var num = parseFloat(val)
  var zd  = isZeroDecimal(getCurrency())
  var err = ''
  if (!val || isNaN(num) || num <= 0) {
    err = 'Please enter a valid number'
  } else if (num < MIN_AMOUNT) {
    err = 'Minimum amount is ' + MIN_AMOUNT.toFixed(zd ? 0 : 2)
  } else if (num > MAX_AMOUNT) {
    err = 'Maximum amount is ' + MAX_AMOUNT.toLocaleString()
  } else if (!zd && !/^\d+(\.\d{1,2})?$/.test(val)) {
    err = 'At most 2 decimal places'
  } else if (zd && !/^\d+$/.test(String(Math.round(num)))) {
    err = getCurrency() + ' does not support decimals'
  }
  // ...
}
```
